    \text{RMSE}(\bar{y}, Y) = \sqrt{\frac{\sum_i^N (y_i - \bar{y})^2}{N}}
\end{equation}

\begin{equation}
    \text{MAE}(\bar{y}, Y) = \frac{\sum_i^N |y_i - \bar{y}|}{N}
\end{equation}

    \text{CRPS}(F,y) =  \int_{-\infty}^{\infty} (F(x) - \mathbb{I}_{x \leq y}(x))^2 dx = \int_{-\infty}^y F(x)^2 dx+ \int_y^\infty(F(x)-1)^2dx
\end{equation}
